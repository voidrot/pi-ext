# Single Pi Extension Config Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Convert this repository into one installable Pi package/extension, `@voidrot/pi-ext`, that hosts multiple internal modules and tools, all controlled by `~/.pi/agent/pi-ext.json[c]` plus trusted project overrides from `.pi/pi-ext.json[c]`.

**Architecture:** Flatten the current `old/` monorepo into a single root package with one Pi manifest entry (`pi.extensions: ["./dist/index.js"]`). The root extension owns config loading, module enable/disable state, tool activation, and `/pi-ext` management commands; internal modules such as DCP register through a small module interface and no-op when disabled. Global config loads first, project config deep-merges on top only when `ctx.isProjectTrusted()` is true.

**Tech Stack:** TypeScript ESM, npm, `tsup`, Node test runner with `tsx`, Pi extension APIs from `@earendil-works/pi-coding-agent`, TypeBox, `@earendil-works/pi-ai`, and `jsonc-parser` for robust JSONC parsing.

---

## Pi Documentation Constraints

Implementation must follow the Pi extension/package docs:

- Extension package exposes exactly one entrypoint through `package.json` `pi.extensions`.
- Runtime dependencies go in `dependencies`; Pi core packages go in `peerDependencies` with `"*"` ranges.
- Use `CONFIG_DIR_NAME` for project `.pi` paths and `getAgentDir()` for `~/.pi/agent`.
- Only honor project-local `.pi/pi-ext.json[c]` when `ctx.isProjectTrusted()` is true.
- `pi.registerTool()` can register tools dynamically, but tools are enabled/disabled through `pi.setActiveTools()`; there is no unregister API.
- Commands and event handlers should be registered once and guard behavior by current module config.
- Long-lived resources must start in session/event handlers, not the extension factory.

---

## Target Config Shape

Support both `.json` and `.jsonc`. Load order is defaults → global `.json` → global `.jsonc` → trusted project `.json` → trusted project `.jsonc`.

Example `~/.pi/agent/pi-ext.jsonc`:

```jsonc
{
  "$schema": "https://raw.githubusercontent.com/voidrot/pi-ext/main/schema/pi-ext.schema.json",
  "enabled": true,
  "debug": false,
  "modules": {
    "dcp": {
      "enabled": true,
      "tools": {
        "compress": { "enabled": true }
      },
      "config": {
        "manualMode": { "enabled": false, "automaticStrategies": true },
        "compress": {
          "permission": "allow",
          "maxContextLimit": "80%",
          "minContextLimit": "40%"
        },
        "strategies": {
          "deduplication": { "enabled": true },
          "purgeErrors": { "enabled": true, "turns": 4 }
        }
      }
    }
  }
}
```

Disable DCP for one project with `.pi/pi-ext.jsonc`:

```jsonc
{
  "modules": {
    "dcp": { "enabled": false }
  }
}
```

Disable only the DCP compression tool:

```jsonc
{
  "modules": {
    "dcp": {
      "tools": { "compress": { "enabled": false } }
    }
  }
}
```

- [ ] **Step 3: Implement config loader**

Create `src/core/config.ts` with:

```typescript
import { existsSync, readFileSync } from "node:fs";
import { join } from "node:path";
import { parse } from "jsonc-parser";
import { CONFIG_DIR_NAME } from "@earendil-works/pi-coding-agent";

export interface ToolToggleConfig {
  enabled: boolean;
}

export interface DcpModuleConfig {
  enabled: boolean;
  tools: { compress: ToolToggleConfig };
  config: Record<string, unknown>;
}

export interface PiExtConfig {
  enabled: boolean;
  debug: boolean;
  modules: { dcp: DcpModuleConfig };
}

export interface LoadPiExtConfigOptions {
  cwd: string;
  agentDir: string;
  projectTrusted: boolean;
}

export const defaultPiExtConfig: PiExtConfig = {
  enabled: true,
  debug: false,
  modules: {
    dcp: {
      enabled: true,
      tools: { compress: { enabled: true } },
      config: {}
    }
  }
};

export function loadPiExtConfig(options: LoadPiExtConfigOptions): PiExtConfig {
  let config = structuredClone(defaultPiExtConfig);
  for (const path of configPaths(options)) {
    if (!existsSync(path)) continue;
    const parsed = parse(readFileSync(path, "utf8"));
    if (!isPlainObject(parsed)) continue;
    config = normalizeConfig(deepMerge(config, parsed));
  }
  return config;
}

function configPaths(options: LoadPiExtConfigOptions): string[] {
  const paths = [
    join(options.agentDir, "pi-ext.json"),
    join(options.agentDir, "pi-ext.jsonc")
  ];
  if (options.projectTrusted) {
    paths.push(
      join(options.cwd, CONFIG_DIR_NAME, "pi-ext.json"),
      join(options.cwd, CONFIG_DIR_NAME, "pi-ext.jsonc")
    );
  }
  return paths;
}

function normalizeConfig(value: unknown): PiExtConfig {
  const merged = deepMerge(defaultPiExtConfig, isPlainObject(value) ? value : {});
  return {
    enabled: toBool(merged.enabled, true),
    debug: toBool(merged.debug, false),
    modules: {
      dcp: {
        enabled: toBool(merged.modules?.dcp?.enabled, true),
        tools: {
          compress: {
            enabled: toBool(merged.modules?.dcp?.tools?.compress?.enabled, true)
          }
        },
        config: isPlainObject(merged.modules?.dcp?.config) ? merged.modules.dcp.config : {}
      }
    }
  };
}

function deepMerge<T>(base: T, override: unknown): T {
  if (!isPlainObject(base) || !isPlainObject(override)) return structuredClone(override as T);
  const output: Record<string, unknown> = { ...base };
  for (const [key, value] of Object.entries(override)) {
    output[key] = isPlainObject(value) && isPlainObject(output[key]) ? deepMerge(output[key], value) : structuredClone(value);
  }
  return output as T;
}

function isPlainObject(value: unknown): value is Record<string, any> {
  return value !== null && typeof value === "object" && !Array.isArray(value);
}

function toBool(value: unknown, fallback: boolean): boolean {
  return typeof value === "boolean" ? value : fallback;
}
```

export interface OwnedTool {
  name: string;
  moduleId: "dcp";
  enabled(config: PiExtConfig): boolean;
}

export const ownedTools: OwnedTool[] = [
  {
    name: "compress",
    moduleId: "dcp",
    enabled: (config) => config.enabled && config.modules.dcp.enabled && config.modules.dcp.tools.compress.enabled
  }
];

export function reconcileActiveTools(activeTools: string[], config: PiExtConfig): string[] {
  const ownedNames = new Set(ownedTools.map((tool) => tool.name));
  const next = new Set(activeTools.filter((name) => !ownedNames.has(name)));
  for (const tool of ownedTools) {
    if (tool.enabled(config)) next.add(tool.name);
  }
  return [...next];
}

```typescript
import { getAgentDir, type ExtensionAPI, type ExtensionContext } from "@earendil-works/pi-coding-agent";
import { loadPiExtConfig, type PiExtConfig } from "./config";
import { Logger } from "./logger";
import { reconcileActiveTools } from "./tools";

export interface PiExtRuntime {
  config: PiExtConfig;
  logger: Logger;
  sessionFile?: string;
}

export function createRuntimeManager(pi: ExtensionAPI) {
  let runtime: PiExtRuntime | undefined;

  function load(ctx: ExtensionContext): PiExtRuntime {
    const sessionFile = ctx.sessionManager.getSessionFile?.();
    if (runtime?.sessionFile === sessionFile) return runtime;

    const config = loadPiExtConfig({
      cwd: ctx.cwd,
      agentDir: getAgentDir(),
      projectTrusted: typeof ctx.isProjectTrusted === "function" ? ctx.isProjectTrusted() : false
    });
    runtime = { config, logger: new Logger(config.debug), sessionFile };
    syncTools();
    return runtime;
  }

  function reload(ctx: ExtensionContext): PiExtRuntime {
    runtime = undefined;
    return load(ctx);
  }

  function syncTools(): void {
    if (!runtime) return;
    pi.setActiveTools(reconcileActiveTools(pi.getActiveTools(), runtime.config));
  }

  return { load, reload, syncTools };
}
```

  return "help";
}

export function registerPiExtCommands(pi: ExtensionAPI, runtime: ReturnTypeOfCreateRuntimeManager): void {
  pi.registerCommand("pi-ext", {
    description: "Show and reload @voidrot/pi-ext configuration",
    handler: async (args, ctx) => {
      const action = parsePiExtCommand(args);
      if (action === "reload") {
        const current = runtime.reload(ctx);
        ctx.ui.notify(`pi-ext config reloaded. enabled=${current.config.enabled}`, "info");
        return;
      }
      const current = runtime.load(ctx);
      if (action === "config") {
        ctx.ui.notify(JSON.stringify(current.config, null, 2), "info");
        return;
      }
      if (action === "status") {
        const dcp = current.config.modules.dcp;
        ctx.ui.notify(
          `pi-ext enabled=${current.config.enabled}; dcp=${dcp.enabled}; compress=${dcp.tools.compress.enabled}`,
          "info"
        );
        return;
      }
      ctx.ui.notify("Usage: /pi-ext [status|config|reload]", "warning");
    }
  });
}
